```gdscript
    # 计算最近敌人距离
    var min_enemy_dist = _get_min_distance_to_enemies(player)
    var avoidance_strength = _cfg(player.player_id).enemy_avoidance_scale
    return avoidance_strength * exp(-3.0 * min_enemy_dist / vision_radius)

func calculate_zone_control_potential(player: Player) -> float:
    # 计算在有利区域的时间
```

```gdscript
# adaptive_reward_system.gd
class_name AdaptiveRewardSystem
var _learning_progress: float = 0.0
var _reward_weights: Dictionary = {
    "explore": 1.0,
    "attack": 1.0,
    "survive": 1.0
}

func update_weights_based_on_progress():
    var progress = _estimate_learning_progress()
    
    if progress < 0.3:  # 早期学习
        _reward_weights["explore"] = 1.5
        _reward_weights["attack"] = 0.8
    elif progress < 0.7:  # 中期学习
        _reward_weights["explore"] = 1.0
        _reward_weights["attack"] = 1.2
    else:  # 后期学习
        _reward_weights["explore"] = 0.6
        _reward_weights["attack"] = 1.5
```

### 技术风险与缓解
1. **算法复杂度风险**：对手建模可能增加训练难度
   - **缓解**：从简单模型开始，逐步增加复杂度
2. **计算资源风险**：多智能体训练需要大量计算
   - **缓解**：优化代码效率，使用分布式训练
3. **收敛风险**：复杂奖励系统可能导致学习不稳定
   - **缓解**：设计渐进式集成方案，充分测试
